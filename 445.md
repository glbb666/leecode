思想：翻转之后相加，用flag记录进位，用sum表示相加之和。

```javascript
function reverse(l){
    if(!l.next)return l;
    var newHead = l;
    var cur = null;
    while(l.next){
        cur = l.next;
        l.next = cur.next;
        cur.next = newHead;
        newHead = cur;
    }
    return newHead;
}
var addTwoNumbers = function(l1, l2) {
    if(!l1)return l2;
    if(!l2)return l1;
    l1 = reverse(l1);
    l2 = reverse(l2);
    var temp = new ListNode(-1);
    var p = temp;
    var flag = 0;
    var sum;
    while(l1||l2){
        sum = 0;
        if(l1)sum+=l1.val
        if(l2)sum+=l2.val;
        sum+=flag;
        sum>=10?flag = 1:flag = 0;
        sum = sum%10;
        p.next = new ListNode(sum);
        p = p.next;
        if(l1)l1 = l1.next;
        if(l2)l2 = l2.next;
    }
    if(flag!=0){
        p.next = new ListNode(flag);
    }
    temp = reverse(temp.next);
    return temp;
};
```
